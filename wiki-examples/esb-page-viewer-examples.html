On Page Config Example

	<div data-esb-page-viewer="http://css-tricks.com"></div>

Triggering loading and unloading of iFrames manually

There is an event-based API to control the loading and unloading of Page Viewer iframes.

  1. Find the UUID of the Page Viewer you want to load/unload
  2. Trigger a custom event on the document using that UUID to invoke the action
				
<div class="wrapper-element">
	<div data-esb-page-viewer="http://example.com"></div>
	<a href="#link" id="unload-iframe-link" class="btn btn-danger">Unload this Page Viewer</a>
	<a href="#link" id="load-iframe-link" class="btn">Load this Page Viewer</a>
</div>	

<!-- First Script Example, Javascript only, see jQuery Example below -->
<script>
	var unload_link = document.getElementById('unload-iframe-link');
	var load_link = document.getElementById('load-iframe-link');

	unload_link.addEventListener('click', function(){
		// The Page Viewer is a sibling of the links in this example
		// Iterate over the siblings until the Page Viewer is found (contains a data-esb-uuid attribute)
		var el = this,
			uuid = false,
			siblings = Array.prototype.filter.call(el.parentNode.children, function(child){
		  		return child !== el;
			});


		Array.prototype.forEach.call(siblings, function(el, i){
			if (el.getAttribute('data-esb-uuid') !== null) {
				uuid = el.getAttribute('data-esb-uuid');
			}
		});

		// Once the uuid is found, dispatch a custom event called 'unload-esb-page-viewer-[uuid]'
		if (uuid) {
			if (window.CustomEvent) {
			  var event = new CustomEvent('unload-esb-page-viewer-' + uuid);
			} else {
			  var event = document.createEvent('CustomEvent');
			  event.initCustomEvent('unload-esb-page-viewer-' + uuid, true, true);
			}

			document.dispatchEvent(event);
		}
	});

	load_link.addEventListener('click', function(){
		// The Page Viewer is a sibling of the links in this example
		// Iterate over the siblings until the Page Viewer is found (contains a data-esb-uuid attribute)
		var el = this,
			uuid = false,
			siblings = Array.prototype.filter.call(el.parentNode.children, function(child){
		  		return child !== el;
			});


		Array.prototype.forEach.call(siblings, function(el, i){
			if (el.getAttribute('data-esb-uuid') !== null) {
				uuid = el.getAttribute('data-esb-uuid');
			}
		});

		// Once the uuid is found, dispatch a custom event called 'unload-esb-page-viewer-[uuid]'
		if (uuid) {
			if (window.CustomEvent) {
			  var event = new CustomEvent('load-esb-page-viewer-' + uuid);
			} else {
			  var event = document.createEvent('CustomEvent');
			  event.initCustomEvent('load-esb-page-viewer-' + uuid, true, true);
			}

			document.dispatchEvent(event);
		}
	});
</script>	

<!-- Second Script Example, jQuery, identical result to above example -->
<script>
	$(document).ready(function(){
		$("#unload-iframe-link").on("click", function(){
			var uuid = $(this).siblings().filter("[data-esb-uuid]").attr("data-esb-uuid");
			$('body').trigger('unload-esb-page-viewer-' + uuid);
		});

		$("#load-iframe-link").on("click", function(){
			var uuid = $(this).siblings().filter("[data-esb-uuid]").attr("data-esb-uuid");
			$(document).trigger('load-esb-page-viewer-' + uuid);
		});
		
	});
</script>
